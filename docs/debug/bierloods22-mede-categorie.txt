  }

  function closeModal() {
    wrapper.classList.remove('is-open');
    resetModal();

  function resetModal() {
    emailInput.value = '';
    orderInput.value = '';
    clearError(emailInput, emailError);
    clearError(orderInput, orderError);
    formSection.style.display = '';
    successSection.style.display = 'none';
    submitBtn.disabled = false;

    input.classList.add('has-error');

    errorEl.style.display = 'block';
  }

  function clearError(input, errorEl) {
    input.classList.remove('has-error');
    errorEl.textContent = '';
    errorEl.style.display = 'none';
  }

  function isValidEmail(val) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val);
  }

  function validate() {
    var valid = true;
    var email = emailInput.value.trim();
    var order = orderInput.value.trim();

    clearError(emailInput, emailError);
    clearError(orderInput, orderError);

    if (!email) {
      showError(emailInput, emailError, 'E-mailadres is vereist');
      valid = false;
    } else if (!isValidEmail(email)) {
      showError(emailInput, emailError, 'Geef een geldig e-mail adres op alstublieft.');
      valid = false;
    }

    if (!order) {
      showError(orderInput, orderError, 'Bestelnummer is verplicht.');
      valid = false;
    }

    return valid;
  }

  function showSuccess() {
    formSection.style.display = 'none';
    successSection.style.display = 'block';
    wrapper.setAttribute('aria-labelledby', 'withdrawal-modal-title-success');
  }

  function submit() {
    if (!validate()) return;

    submitBtn.disabled = true;
    submitBtn.textContent = 'Searching...';

    var formData = new FormData();

      method: 'POST',
      body: formData,
    })
    .then(showSuccess)
    .catch(showSuccess);
  }
